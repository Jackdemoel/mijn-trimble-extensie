/** * focusClash v4.0 * Implementeert handmatige GUID-naar-IDX mapping en meter-conversie. */ async function focusClash(clashItemId) { const item = activeClashItems.find(i => i.id === clashItemId); if (!item || !API) return; // UI Feedback: Markeer actieve rij document.querySelectorAll('.clash-item').forEach(el => el.classList.remove('active')); document.getElementById(`item-${clashItemId}`).classList.add('active'); try { const MODEL_API_URL = "https://model-api21.connect.trimble.com/v1"; const authHeader = { 'Authorization': `Bearer ${accessToken}` }; /** * STAP 1: Handmatige Mapping (Oplossing voor Selectie Bug) * We vragen de interne 'idx' op bij de Model API om de viewer-fallback te voorkomen. */ async function getInternalIdx(modelId, entityId) { try { const url = `${MODEL_API_URL}/models/${modelId}/entities/${entityId}?fields=idx`; const response = await fetch(url, { headers: authHeader }); const data = await response.json(); return data.idx; // Retourneert de integer index (bijv. 38) } catch (e) { console.error("Mapping mislukt voor:", entityId, e); return null; } } const idx1 = await getInternalIdx(item.sourceId1.versionId, item.sourceId1.sourceId); const idx2 = await getInternalIdx(item.sourceId2.versionId, item.sourceId2.sourceId); /** * STAP 2: Selectie opbouwen met 'idx:' prefix * Dit dwingt de viewer om het specifieke object te pakken ipv het hele model. */ const selectionData = []; if (idx1 !== null) { selectionData.push({ modelId: item.sourceId1.versionId, objectIds: ["idx:" + idx1] }); } if (idx2 !== null) { selectionData.push({ modelId: item.sourceId2.versionId, objectIds: ["idx:" + idx2] }); } // Gebruik uitsluitend ondersteunde setSelection methode await API.viewer.setSelection(selectionData); /** * STAP 3: Camera navigatie (Oplossing voor Camera Bug) * Conversie van millimeters (Clash API) naar meters (Viewer API) via factor 0.001. */ const rawCenter = item.center || (item.boundingBox && item.boundingBox.center); if (rawCenter) { const f = 0.001; // Conversie factor const tx = rawCenter.x * f; const ty = rawCenter.y * f; const tz = rawCenter.z * f; // Gebruik uitsluitend de volledige setCamera configuratie om crashes te voorkomen await API.viewer.setCamera({ lookAt: { x: tx, y: ty, z: tz }, position: { x: tx + 0.75, y: ty + 0.75, z: tz + 0.75 }, // Offset voor vogelvlucht projectionType: "perspective" }); } } catch (error) { console.error("Fatale fout in focusClash:", error.message); } }
